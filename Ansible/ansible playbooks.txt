  gather_facts: yes

  tasks:
    - name: install httpd on servers
      action: yum name=httpd state=present


###multiple tasks
[ansible@ip-172-31-45-173 ~]$ cat task.yml
- hosts: devlinux
  user: ansible
  become: yes
  connection: ssh
  gather_facts: yes

  tasks:
    - name: install httpd on servers
      action: yum name=httpd state=present
    - name: start httpd service
      action: service name=httpd state=started
    - name: copy index.html file
      action: copy src=index.html dest=/var/www/html

############################Variables##############

- hosts: devlinux
  user: ansible
  become: yes

  vars:
    pkgname: httpd
    index_path: /var/www/html/

  tasks:
    - name: install apache
      action: yum pkg="{{pkgname}}" state=present
    - name: start apache
      action: service name="{{pkgname}}" state=started
    - name: deploy index file
      action: copy src=index.html dest="{{index_path}}"


#################handlers#############

- hosts: devlinux
  user: ansible
  connection: ssh
  become: yes

  tasks:
    - name: install httpd
      action: yum pkg="httpd" state="present"
      notify: start service

  handlers:
        - name: start service
          action: service name=httpd state=started
		  
### multiple handlers


- hosts: devlinux
  user: ansible
  connection: ssh
  become: yes

  tasks:
    - name: install httpd
      action: yum pkg="httpd" state="present"
      notify:
        - start
        - deploy

  handlers:
        - name: start
          action: service name=httpd state=started
        - name: deploy
          action: copy src=index.html dest=/var/www/html
